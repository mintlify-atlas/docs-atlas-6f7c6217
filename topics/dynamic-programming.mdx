Dynamic Programming is the single most-tested topic in coding interviews at top companies. The repository contains **150+ DP solutions** organized into focused sub-categories so you can study patterns systematically rather than treating each problem as a one-off.

<Note>
  The standard progression for any DP problem is: **recursion → memoization (top-down) → tabulation (bottom-up) → space optimization**. Always start with the recursive solution to nail the recurrence, then optimize.
</Note>

## Code: Climbing Stairs

**Company Tags:** Google, Amazon, OYO Rooms, Microsoft, Adobe, Flipkart, Siemens

Three progressively optimized approaches — all from the repo.

<Tabs>
  <Tab title="Memoization (Top-Down)">
    ```cpp
    // T.C: O(n)  S.C: O(n)
    class Solution {
    public:
        int t[46];
        int solve(int n) {
            if (n == 0 || n == 1 || n == 2)
                return n;
            if (t[n] != -1)
                return t[n];
            return t[n] = solve(n - 1) + solve(n - 2);
        }
        int climbStairs(int n) {
            memset(t, -1, sizeof(t));
            return solve(n);
        }
    };
    ```
  </Tab>
  <Tab title="Bottom-Up Tabulation">
    ```cpp
    // T.C: O(n)  S.C: O(n)
    class Solution {
    public:
        int climbStairs(int n) {
            if (n == 1 || n == 2) return n;
            vector<int> t(n + 1);
            t[0] = 0; t[1] = 1; t[2] = 2;
            for (int i = 3; i <= n; i++)
                t[i] = t[i - 1] + t[i - 2];
            return t[n];
        }
    };
    ```
  </Tab>
  <Tab title="Space-Optimized O(1)">
    ```cpp
    // T.C: O(n)  S.C: O(1)
    class Solution {
    public:
        int climbStairs(int n) {
            if (n == 1 || n == 2 || n == 3) return n;
            int a = 1, b = 2, c = 3;
            for (int i = 3; i <= n; i++) {
                c = a + b;
                int temp_b = b;
                b = c;
                a = temp_b;
            }
            return c;
        }
    };
    ```
  </Tab>
</Tabs>

## Code: Longest Common Subsequence

**Company Tags:** Microsoft, Amazon, FactSet, Hike, Citrix, MakeMyTrip, Paytm, VMWare

<Tabs>
  <Tab title="Memoization (Top-Down)">
    ```cpp
    // T.C: O(m*n)  S.C: O(m*n)
    class Solution {
    public:
        int t[1001][1001];
        int LCS(string& s1, string& s2, int m, int n) {
            if (m == 0 || n == 0)
                return t[m][n] = 0;
            if (t[m][n] != -1)
                return t[m][n];
            if (s1[m - 1] == s2[n - 1])
                return t[m][n] = 1 + LCS(s1, s2, m - 1, n - 1);
            return t[m][n] = max(LCS(s1, s2, m, n - 1), LCS(s1, s2, m - 1, n));
        }
        int longestCommonSubsequence(string text1, string text2) {
            int m = text1.length(), n = text2.length();
            memset(t, -1, sizeof(t));
            return LCS(text1, text2, m, n);
        }
    };
    ```
  </Tab>
  <Tab title="Bottom-Up Tabulation">
    ```cpp
    // T.C: O(m*n)  S.C: O(m*n)
    class Solution {
    public:
        int longestCommonSubsequence(string text1, string text2) {
            int m = text1.length(), n = text2.length();
            vector<vector<int>> t(m + 1, vector<int>(n + 1));
            for (int i = 0; i <= m; i++) {
                for (int j = 0; j <= n; j++) {
                    if (i == 0 || j == 0)
                        t[i][j] = 0;
                    else if (text1[i - 1] == text2[j - 1])
                        t[i][j] = 1 + t[i - 1][j - 1];
                    else
                        t[i][j] = max(t[i][j - 1], t[i - 1][j]);
                }
            }
            return t[m][n];
        }
    };
    ```
  </Tab>
  <Tab title="Space-Optimized O(n)">
    ```cpp
    // T.C: O(m*n)  S.C: O(n)  — only two rows needed
    class Solution {
    public:
        int longestCommonSubsequence(string text1, string text2) {
            int m = text1.length(), n = text2.length();
            vector<vector<int>> t(2, vector<int>(n + 1));
            for (int i = 0; i <= m; i++) {
                for (int j = 0; j <= n; j++) {
                    if (i == 0 || j == 0)
                        t[i % 2][j] = 0;
                    else if (text1[i - 1] == text2[j - 1])
                        t[i % 2][j] = 1 + t[(i + 1) % 2][j - 1];
                    else
                        t[i % 2][j] = max(t[i % 2][j - 1], t[(i + 1) % 2][j]);
                }
            }
            return t[m % 2][n];
        }
    };
    ```
  </Tab>
</Tabs>

## Study Tips

<Tip>
  Always derive the recurrence on paper first. If you cannot express `f(n)` in terms of smaller subproblems, you are not ready to code yet.
</Tip>

<Tip>
  For any new DP problem: identify **what changes** between recursive calls — that is your state. Anything that changes is a dimension of your DP table.
</Tip>

<Tip>
  Space optimization is almost always possible when `dp[i]` depends only on `dp[i-1]` (1-D → O(1)) or when a 2-D table only reads the previous row (2-D → O(n)).
</Tip>
